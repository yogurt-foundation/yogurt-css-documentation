```scss
// Add Values

  $class: 'shadow-md-gray',

  $modifier: (
    100: 0 0 6px #262626,

  )

@include yogurt(
  $class: 'shadow-md-gray',

  ),
  $variant: (
    'dark-mode',
    'focus-within', 'focus-visible'
    //...
  )
);
```
